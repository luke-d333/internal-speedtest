<body>

  <h1>Internal Speedtest</h1>

  <button id="run">Run Speed Test</button>

  <div id="results">Click the button to start the test.</div>


  <script>

    document.getElementById("run").onclick = async () => {

      const resultsDiv = document.getElementById("results");

      resultsDiv.innerText = "Running test...";

      

      try {

        const res = await fetch("YOUR_MAKE_WEBHOOK_URL");

        const data = await res.json();

        

        resultsDiv.innerText = 

          `Download: ${data.download} Mbps\n` +

          `Upload: ${data.upload} Mbps\n` +

          `Latency: ${data.latency} ms`;

      } catch (err) {

        resultsDiv.innerText = "Error running the speed test. Check the webhook URL.";

      }

    }

  </script>
